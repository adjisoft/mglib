517template <typename T, std::size_t N>
518MG_NODISCARD MG_INLINE auto determinant(const Mat<T, N, N>& matrix) -> detail::compute_type_t<T> {
519 using Compute = detail::compute_type_t<T>;
520 Mat<Compute, N, N> work;
521 for (std::size_t row = 0; row < N; ++row) {
522 for (std::size_t column = 0; column < N; ++column) {
523 work(row, column) = static_cast<Compute>(matrix(row, column));
524 }
525 }
526
527 Compute det = static_cast<Compute>(1);
528 int sign = 1;
529
530 for (std::size_t pivot = 0; pivot < N; ++pivot) {
531 std::size_t pivot_row = pivot;
532 Compute pivot_value = detail::abs_value(work(pivot_row, pivot));
533 for (std::size_t row = pivot + 1; row < N; ++row) {
534 const Compute candidate = detail::abs_value(work(row, pivot));
535 if (candidate > pivot_value) {
536 pivot_value = candidate;
537 pivot_row = row;
538 }
539 }
540
541 if (detail::near_zero(pivot_value)) {
542 return Compute{};
543 }
544
545 if (pivot_row != pivot) {
546 for (std::size_t column = 0; column < N; ++column) {
547 std::swap(work(pivot, column), work(pivot_row, column));
548 }
549 sign *= -1;
550 }
551
552 const Compute diagonal = work(pivot, pivot);
553 det *= diagonal;
554 for (std::size_t row = pivot + 1; row < N; ++row) {
555 const Compute factor = work(row, pivot) / diagonal;
556 work(row, pivot) = Compute{};
557 for (std::size_t column = pivot + 1; column < N; ++column) {
558 work(row, column) -= factor * work(pivot, column);
559 }
560 }
561 }
562
563 return sign > 0 ? det : -det;
564}

566template <typename T, std::size_t N>
567MG_NODISCARD MG_INLINE auto inverse(const Mat<T, N, N>& matrix)
568 -> Mat<detail::compute_type_t<T>, N, N> {
569 using Compute = detail::compute_type_t<T>;
570 Mat<Compute, N, N> work;
571 Mat<Compute, N, N> inverse_matrix = Mat<Compute, N, N>::identity();
572
573 for (std::size_t row = 0; row < N; ++row) {
574 for (std::size_t column = 0; column < N; ++column) {
575 work(row, column) = static_cast<Compute>(matrix(row, column));
576 }
577 }
578
579 for (std::size_t pivot = 0; pivot < N; ++pivot) {
580 std::size_t pivot_row = pivot;
581 Compute pivot_value = detail::abs_value(work(pivot_row, pivot));
582 for (std::size_t row = pivot + 1; row < N; ++row) {
583 const Compute candidate = detail::abs_value(work(row, pivot));
584 if (candidate > pivot_value) {
585 pivot_value = candidate;
586 pivot_row = row;
587 }
588 }
589
590 if (detail::near_zero(pivot_value)) {
591 throw std::runtime_error("mg::math::inverse cannot invert a singular matrix");
592 }
593
594 if (pivot_row != pivot) {
595 for (std::size_t column = 0; column < N; ++column) {
596 std::swap(work(pivot, column), work(pivot_row, column));
597 std::swap(inverse_matrix(pivot, column), inverse_matrix(pivot_row, column));
598 }
599 }
600
601 const Compute diagonal = work(pivot, pivot);
602 for (std::size_t column = 0; column < N; ++column) {
603 work(pivot, column) /= diagonal;
604 inverse_matrix(pivot, column) /= diagonal;
605 }
606
607 for (std::size_t row = 0; row < N; ++row) {
608 if (row == pivot) {
609 continue;
610 }
611 const Compute factor = work(row, pivot);
612 if (detail::near_zero(factor)) {
613 continue;
614 }
615 for (std::size_t column = 0; column < N; ++column) {
616 work(row, column) -= factor * work(pivot, column);
617 inverse_matrix(row, column) -= factor * inverse_matrix(pivot, column);
618 }
619 }
620 }
621
622 return inverse_matrix;
623}

642 for (std::size_t index = 0; index < N; ++index) {
643 result.pivot_indices[index] = index;
644 for (std::size_t column = 0; column < N; ++column) {
645 result.upper(index, column) = static_cast<Compute>(matrix(index, column));
646 }
647 }
648
649 for (std::size_t pivot = 0; pivot < N; ++pivot) {
650 std::size_t pivot_row = pivot;
651 Compute pivot_value = detail::abs_value(result.upper(pivot_row, pivot));
652 for (std::size_t row = pivot + 1; row < N; ++row) {
653 const Compute candidate = detail::abs_value(result.upper(row, pivot));
654 if (candidate > pivot_value) {
655 pivot_value = candidate;
656 pivot_row = row;
657 }
658 }
659
660 if (detail::near_zero(pivot_value)) {
661 return result;
662 }
663
664 if (pivot_row != pivot) {
665 for (std::size_t column = 0; column < N; ++column) {
666 std::swap(result.upper(pivot, column), result.upper(pivot_row, column));
667 std::swap(result.permutation(pivot, column), result.permutation(pivot_row, column));
668 }
669 for (std::size_t column = 0; column < pivot; ++column) {
670 std::swap(result.lower(pivot, column), result.lower(pivot_row, column));
671 }
672 std::swap(result.pivot_indices[pivot], result.pivot_indices[pivot_row]);
673 result.permutation_sign *= -1;
674 }
675
676 for (std::size_t row = pivot + 1; row < N; ++row) {
677 const Compute factor = result.upper(row, pivot) / result.upper(pivot, pivot);
678 result.lower(row, pivot) = factor;
679 result.upper(row, pivot) = Compute{};
680 for (std::size_t column = pivot + 1; column < N; ++column) {
681 result.upper(row, column) -= factor * result.upper(pivot, column);
682 }
683 }
684 }
685
686 result.success = true;
687 return result;
688}

690template <typename T, std::size_t Rows, std::size_t Cols>
691struct QRResult {
692 using value_type = detail::compute_type_t<T>;
693
694 Mat<value_type, Rows, Cols> q{};
695 Mat<value_type, Cols, Cols> r{};
696 bool success = false;
697};

699template <typename T, std::size_t Rows, std::size_t Cols>
700MG_NODISCARD MG_INLINE auto qr_decompose(const Mat<T, Rows, Cols>& matrix) -> QRResult<T, Rows, Cols> {
701 using Compute = detail::compute_type_t<T>;
702 QRResult<T, Rows, Cols> result;
703
704 for (std::size_t column = 0; column < Cols; ++column) {
705 Vec<Compute, Rows> working_column;
706 for (std::size_t row = 0; row < Rows; ++row) {
707 working_column[row] = static_cast<Compute>(matrix(row, column));
708 }
709
710 for (std::size_t previous = 0; previous < column; ++previous) {
711 Vec<Compute, Rows> q_column;
712 for (std::size_t row = 0; row < Rows; ++row) {
713 q_column[row] = result.q(row, previous);
714 }
715 result.r(previous, column) = dot(q_column, working_column);
716 working_column = working_column - q_column * result.r(previous, column);
717 }
718
719 const Compute column_norm = norm(working_column);
720 if (detail::near_zero(column_norm)) {
721 return result;
722 }
723
724 result.r(column, column) = column_norm;
725 const auto normalized = working_column / column_norm;
726 for (std::size_t row = 0; row < Rows; ++row) {
727 result.q(row, column) = normalized[row];
728 }
729 }
730
731 result.success = true;
732 return result;
733}

735template <typename T, std::size_t N>
736struct PowerIterationResult {
737 using value_type = detail::compute_type_t<T>;
738
739 value_type eigenvalue{};
740 Vec<value_type, N> eigenvector{};
741 std::size_t iterations = 0;
742 bool success = false;
743};
744
745template <typename T, std::size_t N>
746MG_NODISCARD MG_INLINE auto power_iteration(const Mat<T, N, N>& matrix,
747 std::size_t max_iterations = 256,
748 detail::compute_type_t<T> tolerance =
749 mg::epsilon_v<detail::compute_type_t<T>>)
750 -> PowerIterationResult<T, N> {
751 using Compute = detail::compute_type_t<T>;
752 PowerIterationResult<T, N> result;
753
754 Vec<Compute, N> vector(static_cast<Compute>(1));
755 vector = normalize(vector);
756
757 for (std::size_t iteration = 0; iteration < max_iterations; ++iteration) {
758 const auto next = matrix * vector;
759 const Compute next_norm = norm(next);
760 if (detail::near_zero(next_norm)) {
761 return result;
762 }
763
764 const auto normalized = next / next_norm;
765 if (norm(normalized - vector) <= tolerance) {
766 vector = normalized;
767 result.iterations = iteration + 1;
768 result.success = true;
769 break;
770 }
771
772 vector = normalized;
773 result.iterations = iteration + 1;
774 }
775
776 const auto transformed = matrix * vector;
777 result.eigenvalue = dot(vector, transformed) / dot(vector, vector);
778 result.eigenvector = vector;
779 return result;
780}

866 }
