101template <typename T>
102MG_INLINE void append_value(std::ostream& output, const T& value) {
103 if constexpr (std::is_same<decay_t<T>, bool>::value) {
104 output << (value ? "true" : "false");
105 } else if constexpr (is_string_like<T>::value) {
106 output << value;
107 } else if constexpr (is_pair<decay_t<T>>::value) {
108 append_pair(output, value);
109 } else if constexpr (is_iterable<T>::value && !is_string_like<T>::value && !is_path<T>::value) {
110 append_iterable(output, value);
111 } else if constexpr (is_streamable<T>::value) {
112 output << value;
113 } else {
114 static_assert(always_false<T>::value, "mg::fmt::format cannot format this type");
115 }
116}
